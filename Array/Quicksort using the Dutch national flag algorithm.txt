void quicksort(vector<int> &arr,int low,int high){
    if(low<high){
        int pivot = partition(arr,low,high);
        quicksort(arr,low,pivot-1);
        quicksort(arr,pivot+1,high);
    }
}
    int partition(vector<int>&arr,int low,int high){
        int i = low;
        int j = high;
        int pivot = arr[low];
        
        while(i<j){
            while(arr[i]<=pivot){
                i++;
            }
            while(arr[j]>pivot){
                j--;
            }
            if(i<j){
                swap(arr[i],arr[j]);
            }
        }
        swap(arr[low],arr[j]);
        return j;
    }
